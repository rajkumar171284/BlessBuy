let count = 0;
for (const product of sampleProducts) {
  // Simulate form fill and submit
  const form = document.querySelector('form');
  if (form) {
    // Fill name
    const nameInput = form.querySelector('input[placeholder*="name"]') || form.querySelector('input[type="text"]:first-child');
    if (nameInput) nameInput.value = product.name;
    
    // Fill price
    const priceInput = form.querySelector('input[type="number"][placeholder*="price"]') || form.querySelector('input[type="number"]:first-child');
    if (priceInput) priceInput.value = product.price;
    
    count++;
  }
}
